scrape_configs:
  - job_name: 'ec2_sd'
    ec2_sd_configs:
      - region: us-east-1
        port: 9100  # The port your exporter (e.g., node_exporter) is listening on
        filters:
          - name: "tag:Monitoring"
            values: ['true']
          - name: "instance-state-name"
            values: ['running']
    
    relabel_configs:
      # Use the EC2 'Name' tag as the 'instance' label in Prometheus
      - source_labels: [__meta_ec2_instance_id]
        target_label: instance_id
      
      # Use the private IP for scraping (default behavior, but can be explicitly set)
      - source_labels: [__meta_ec2_tag_Name]
        target_label: app
      - source_labels: [__meta_ec2_private_ip]
